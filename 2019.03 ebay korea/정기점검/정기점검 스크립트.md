```
function hr() {  
    HR="==========================================================================="  
    if [[ $1 != "" ]]; then  
    echo -e "\n\n\n$HR"  
        echo "| $1"  
    fi  
    echo "$HR"  
}
```
 
```
function printout() {  
    if [[ $2 == *$'\n'* ]]; then  
        echo "$1"  
        echo "$2"  
        echo  
    else  
        printf "%-20s%s\n" "$1" "$2"  
    fi
```
 
```
    echo '---------------------------------------------------------------------------'  
}
```
   

```
#==========================================================================
```
 
```
hr "OS & Kernel"
```
   

```
RHEL_VERSION=$(cat /etc/redhat-release)  
printout RHEL_VERSION "$RHEL_VERSION"
```
    
```
KERNEL_VERSION=$(uname -r)  
printout KERNEL_VERSION "$KERNEL_VERSION"
```
    
```
UPTIME=$(uptime | sed -Ee 's/.+up (([0-9:]+[ a-z]*)+), .+/\1/')  
printout UPTIME "$UPTIME"
```
    
```
hr "CPU"
```
   

```
CPU_NUM_CORE=$(cat /proc/cpuinfo | grep processor | wc -l)  
printout CPU_NUM_CORE "$CPU_NUM_CORE"  
CPU_UTILIZATION=$(sar 5 10 | grep Average | awk '{print 100 - $8}')  
printout CPU_UTILIZATION "$CPU_UTILIZATION"
```
      

```
hr "Memory"
```
   

```
MEM_INFO=$(free -m | grep Mem)
```
 
```
MEM_TOTAL=$(echo $MEM_INFO | awk '{print $2/1024 " GB"}')  
printout MEM_TOTAL "$MEM_TOTAL"
```
 
```
MEM_USED=$(echo $MEM_INFO | awk '{print $3/1024 " GB"}')  
printout MEM_USED "$MEM_USED"
```
 
```
MEM_FREE=$(echo $MEM_INFO | awk '{print $4/1024 " GB"}')  
printout MEM_FREE "$MEM_FREE"
```
 
```
MEM_AVAILABLE=$(echo $MEM_INFO | awk '{print $7/1024 " MB"}')  
printout MEM_AVAILABLE "$MEM_AVAILABLE"
```
    
```
hr "Journal errors"  
journalctl -p3 | grep ''
```
